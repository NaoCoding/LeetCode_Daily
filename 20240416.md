class Solution:
    def addOneRow(self, root: Optional[TreeNode], val: int, depth: int) -> Optional[TreeNode]:
        if depth == 1:
            fn = TreeNode(val)
            fn.left = root
            return fn
        wait = [root]
        n = 1
        while n + 1 < depth:
            q = []
            for j in wait:
                if j.left:
                    q.append(j.left)
                if j.right:
                    q.append(j.right)
            if len(q):
                wait = q
            n += 1
        for j in wait:
            if j.left:
                r = TreeNode(val)
                r.left = j.left
                j.left = r
            else:
                r = TreeNode(val)
                j.left = r
            if j.right:
                r = TreeNode(val)
                r.right = j.right
                j.right = r
            else:
                r = TreeNode(val)
                j.right = r
        return root
